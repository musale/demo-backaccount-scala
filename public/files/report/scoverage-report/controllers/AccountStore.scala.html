115   def readFromFile(): List[AccountData] = {
116     import scala.collection.mutable.ListBuffer
117     try {
118       val account_data = Source.fromFile(filename)
119       var account_list = new ListBuffer[AccountData]()
120       for (line <- account_data.getLines()) {
121         Json.parse(line).validate[AccountData] match {
122           case s: JsSuccess[AccountData] => {
123             val acc: AccountData = s.get
124             account_list += acc
125           }
126           case e: JsError => {
127             logger.error("error reading from file: %s", e.get)
128           }
129         }
130       }
131       account_list.toList
132     } catch {
133       case ex: FileNotFoundException => {
134         new FileOutputStream(filename, true)
135         val account_data = Source.fromFile(filename)

157   // Make a deposit
158   override def deposit(amount: AccountAmount)(
159       implicit mc: MarkerContext): Future[Response] = {
160     Future {
161       logger.trace(s"deposit: amount = $amount")
162       // get the current transaction from the list
163       val transactions = readFromFile()
164 
165       var previous_amount: Double = 0.0
166       val timestamp: Long = Instant.now.getEpochSecond
167       if (!(transactions.isEmpty)) {
168         val previous: AccountData = transactions.last
169         previous_amount = AccountAmount(previous.amount).toDouble
170       }
171 
172       // Max daily deposit is USD 150,000
173       val MAX_DEPOSIT = 150000.toDouble
174       val todays_deposits = transactions
175         .filter(_.ttype == "Deposit")
176         .filter(_.timestamp <= timestamp)
177         .filter(_.timestamp >= midnightstamp)
178       var todays_deposits_total: Double = 0.0
179       if (!(todays_deposits.isEmpty)) {
180         todays_deposits_total = todays_deposits.last.amount.toDouble
181       }
182       // Max transactions per day is 4
183       val MAX_TRANSACTIONS = 4
184       val total_transactions = todays_deposits.length
185       // Max input is USD 40,000
186       val MAX_INPUT = 40000.toDouble
187       // Amount should be string
188       val input = amount.toDouble
189 
190       // deposit a max of 40k per transaction
191       if (input > MAX_INPUT) {
192         Response("error",
193                  s"exceeded maximum deposit of USD $MAX_INPUT per transaction",
194                  403)
195       }
196       // deposit a total max of 150k per day
197       else if ((todays_deposits_total + input) >= MAX_DEPOSIT) {
198         Response("error",
199                  s"exceeded maximum deposit amount of USD $MAX_DEPOSIT per day",
200                  403)
201       }
202       // deposit in a max of 4 transations
203       else if (total_transactions >= MAX_TRANSACTIONS) {
204         Response(
205           "error",
206           s"exceeded maximum deposit frequency of USD $MAX_TRANSACTIONS times per day",
207           403)
208       }
209       // all conditions passed
210       else {
211         val current: Double = previous_amount + input
212         val account = AccountData(current.toString, timestamp, "Deposit")
213         // Add the transaction to transactions
214         writeTofile(account)
215         Response("OK", s"new balance after deposit is USD $current", 201)
216       }
217     }
218   }

236         .filter(_.ttype == "Withdraw")
237         .filter(_.timestamp <= timestamp)
238         .filter(_.timestamp >= midnightstamp)

240 
241       // Last deposit made

244 
245       if (!(todays_withdrawals.isEmpty)) {
246         todays_withdraw_total = last_deposit - todays_withdrawals.last.amount.toDouble
247       }
248 
249       // Max transactions per day is 3
250       val MAX_TRANSACTIONS = 3
251       val total_transactions = todays_withdrawals.length
252       val input = amount.toDouble

254 
255       // Max input is USD 20,000
256       val MAX_INPUT = 20000.toDouble
257 
258       // withdrawal amount should be less than current balance
259       if (input > previous_amount) {
260         Response("error",
261                  "account balance is lower than withdrawal amount",
262                  403)
263       }
264       // withdraw less than 20K in a single transaction
265       else if (input > MAX_INPUT) {
266         Response(
267           "error",
268           s"exceeded maximum withdrawal of USD $MAX_INPUT per transaction",
269           403)
270       }
271       // withdraw a total maximum of 50k per day

273         Response(
274           "error",
275           s"exceeded maximum withdrawal of USD $MAX_WITHDRAWAL amount per day",
276           403)
277       }

283           403)
284       }
